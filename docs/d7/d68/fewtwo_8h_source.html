183  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
184  {
204  // start the clock
205  timer.Reset();
206 
207  // define terminals based on size of X
208  params.set_terminals(X.rows());
209 
210  // initial model on raw input
211  params.msg("Fitting initial model", 1);
212  initial_model(X,y);
213 
214  // initialize population
215  params.msg("Initializing population", 1);
216  p_pop->init(best_ind,params);
217 
218  // resize F to be twice the pop-size x number of samples
219  F.resize(X.cols(),int(2*params.pop_size));
220 
221  // evaluate initial population
222  params.msg("Evaluating initial population",1);
223  p_eval->fitness(*p_pop,X,y,F,params);
224 
225  vector<size_t> survivors;
226 
227  // main generational loop
228  for (unsigned int g = 0; g<params.gens; ++g)
229  {
230 
231  // select parents
232  params.msg("selection..", 2);
233  vector<size_t> parents = p_sel->select(F, params, r);
234  params.msg("parents:\n"+p_pop->print_eqns(","), 2);
235 
236  // variation to produce offspring
237  params.msg("variation...", 2);
238  p_variation->vary(*p_pop, parents, params);
239  params.msg("offspring:\n" + p_pop->print_eqns(true), 2);
240 
241  // evaluate offspring
242  params.msg("evaluating offspring...", 2);
243  p_eval->fitness(*p_pop, X, y, F, params, true);
244 
245  // select survivors from combined pool of parents and offspring
246  params.msg("survival", 2);
247  survivors = p_surv->select(*p_pop, params, r);
248 
249  // reduce population to survivors
250  p_pop->update(survivors);
251  params.msg("survivors:\n" + p_pop->print_eqns(), 2);
252 
253  update_best();
254  if (params.verbosity>0) print_stats(g+1);
255  }
256  params.msg("finished",1);
257  params.msg("best representation: " + best_ind.get_eqn(),1);
258  params.msg("score: " + std::to_string(best_score), 1);
259  }

261  void Fewtwo::initial_model(MatrixXd& X, VectorXd& y)
262  {
266  bool pass = true;
267  VectorXd yhat = p_eval->out_ml(X,y,params,pass,p_ml);
268 
269  // set terminal weights based on model
270  params.set_term_weights(p_ml->get_weights());
271 
272  // assign best score as MSE
273  best_score = (yhat-y).array().pow(2).mean();
274 
275  // initialize best_ind to be all the features
276  best_ind = Individual();
277  for (unsigned i =0; i<X.rows(); ++i)
278  best_ind.program.push_back(params.terminals[i]);
279  best_ind.fitness = best_score;
280  }
