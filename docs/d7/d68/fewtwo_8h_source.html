241  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
242  {
262  // start the clock
263  timer.Reset();
264 
265  // split data into training and test sets
266  MatrixXd X_t(X.rows(),int(X.cols()*params.split));
267  MatrixXd X_v(X.rows(),int(X.cols()*(1-params.split)));
268  VectorXd y_t(int(y.size()*params.split)), y_v(int(y.size()*(1-params.split)));
269  train_test_split(X,y,X_t,X_v,y_t,y_v,params.shuffle);
270 
271  // define terminals based on size of X
272  params.set_terminals(X.rows());
273 
274  // initial model on raw input
275  params.msg("Fitting initial model", 1);
276  initial_model(X,y);
277  params.msg("Initial score: " + std::to_string(best_score), 1);
278 
279  // initialize population
280  params.msg("Initializing population", 1);
281  p_pop->init(best_ind,params);
282  params.msg("Initial population:\n"+p_pop->print_eqns(),2);
283 
284  // resize F to be twice the pop-size x number of samples
285  F.resize(X_t.cols(),int(2*params.pop_size));
286 
287  // evaluate initial population
288  params.msg("Evaluating initial population",1);
289  p_eval->fitness(*p_pop,X_t,y_t,F,params);
290 
291  vector<size_t> survivors;
292 
293  // main generational loop
294  for (unsigned int g = 0; g<params.gens; ++g)
295  {
296 
297  // select parents
298  params.msg("selection..", 2);
299  vector<size_t> parents = p_sel->select(*p_pop, F, params);
300  params.msg("parents:\n"+p_pop->print_eqns(), 2);
301 
302  // variation to produce offspring
303  params.msg("variation...", 2);
304  p_variation->vary(*p_pop, parents, params);
305  params.msg("offspring:\n" + p_pop->print_eqns(true), 2);
306 
307  // evaluate offspring
308  params.msg("evaluating offspring...", 2);
309  p_eval->fitness(*p_pop, X_t, y_t, F, params, true);
310 
311  // select survivors from combined pool of parents and offspring
312  params.msg("survival...", 2);
313  survivors = p_surv->survive(*p_pop, F, params);
314 
315  // reduce population to survivors
316  params.msg("shrinking pop to survivors...",2);
317  p_pop->update(survivors);
318  params.msg("survivors:\n" + p_pop->print_eqns(), 2);
319 
320  update_best();
321  if (params.verbosity>0) print_stats(g+1);
322  }
323  params.msg("finished",1);
324  params.msg("best training representation: " + best_ind.get_eqn(),1);
325  params.msg("train score: " + std::to_string(best_score), 1);
326  // evaluate population on validation set
327  F_v.resize(X_v.cols(),int(2*params.pop_size));
328  p_eval->fitness(*p_pop, X_v, y_v, F_v, params);
329  initial_model(X_v, y_v); // calculate baseline model validation score
330  update_best(); // get the best validation model
331  params.msg("best validation representation: " + best_ind.get_eqn(),1);
332  params.msg("validation score: " + std::to_string(best_score), 1);
333  }

335  void Fewtwo::initial_model(MatrixXd& X, VectorXd& y)
336  {
340  bool pass = true;
341  VectorXd yhat = p_eval->out_ml(X,y,params,pass,p_ml);
342 
343  // set terminal weights based on model
344  params.set_term_weights(p_ml->get_weights());
345 
346  if (params.classification) // assign best score as mean accuracy
347  best_score = (yhat.cast<int>().array() != y.cast<int>().array()).cast<double>().mean();
348  else // assign best score as MSE
349  best_score = (yhat-y).array().pow(2).mean();
350 
351  // initialize best_ind to be all the features
352  best_ind = Individual();
353  for (unsigned i =0; i<X.rows(); ++i)
354  best_ind.program.push_back(params.terminals[i]);
355  best_ind.fitness = best_score;
356  }

395  void Fewtwo::print_stats(unsigned int g)
396  {
397  unsigned num_models = std::min(20,p_pop->size());
398  double med_score = median(F.colwise().mean().array()); // median loss
399  ArrayXd Sizes(p_pop->size()); unsigned i = 0; // collect program sizes
400  for (const auto& p : p_pop->individuals){ Sizes(i) = p.size(); ++i;}
401  unsigned med_size = median(Sizes); // median program size
402  unsigned max_size = Sizes.maxCoeff();
403  string bar, space = ""; // progress bar
404  for (unsigned int i = 0; i<50; ++i){
405  if (i <= 50*g/params.gens) bar += "/";
406  else space += " ";
407  }
408  std::cout.precision(5);
409  std::cout << std::scientific;
410  std::cout << "Generation " << g << "/" << params.gens << " [" + bar + space + "]\n";
411  std::cout << "Min Loss\tMedian Loss\tMedian (Max) Size\tTime (s)\n"
412  << best_score << "\t" << med_score << "\t" ;
413  std::cout << std::fixed << med_size << " (" << max_size << ") \t\t" << timer << "\n";
414  std::cout << "Representation Pareto Front--------------------------------------\n";
415  std::cout << "Rank\tComplexity\tLoss\tRepresentation\n";
416  std::cout << std::scientific;
417  // printing 10 individuals from the pareto front
418  unsigned n = 1;
419  vector<size_t> f = p_pop->sorted_front(n);
420  vector<size_t> fnew(2,0);
421  while (f.size() < num_models && fnew.size()>1)
422  {
423  fnew = p_pop->sorted_front(++n);
424  f.insert(f.end(),fnew.begin(),fnew.end());
425  }
426 
427  for (unsigned j = 0; j < std::min(num_models,unsigned(f.size())); ++j)
428  {
429  std::cout << p_pop->individuals[f[j]].rank << "\t"
430  << p_pop->individuals[f[j]].complexity() << "\t" << (*p_pop)[f[j]].fitness
431  << "\t" << p_pop->individuals[f[j]].get_eqn() << "\n";
432  }
433 
434 
435  // ref counting
436  vector<float> use(params.terminals.size());
437  float use_sum=0;
438  for (unsigned i = 0; i< params.terminals.size(); ++i)
439  {
440  use[i] = float(params.terminals[i].use_count());
441  use_sum += use[i];
442  }
443  vector<size_t> use_idx = argsort(use);
444  std::reverse(use_idx.begin(), use_idx.end());
445 
446  int nf = std::min(5,int(params.terminals.size()));
447  std::cout << "Top " << nf <<" features (\% usage):\n";
448  std::cout.precision(1);
449  for (unsigned i = 0; i<nf; ++i)
450  std::cout << std::fixed << params.terminals[use_idx[i]]->name << "_"
451  << std::dynamic_pointer_cast<NodeVariable>(params.terminals[use_idx[i]])->loc
452  << " (" << use[use_idx[i]]/use_sum*100 << "\%)\t";
453 
454  std::cout <<"\n\n";
455  }
