254  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
255  {
275  // start the clock
276  timer.Reset();
277 
278  if(str_dim.compare("") != 0)
279  {
280  string dimension;
281  dimension = str_dim.substr(0, str_dim.length() - 1);
282  params.msg("STR DIM IS "+ dimension, 1);
283  params.msg("Cols are " + std::to_string(X.cols()), 1);
284  params.msg("Setting dimensionality as " + std::to_string((int)(ceil(stod(dimension)*X.cols()))), 1);
285  set_max_dim(ceil(stod(dimension)*X.cols()));
286  }
287 
288  // split data into training and test sets
289  MatrixXd X_t(X.rows(),int(X.cols()*params.split));
290  MatrixXd X_v(X.rows(),int(X.cols()*(1-params.split)));
291  VectorXd y_t(int(y.size()*params.split)), y_v(int(y.size()*(1-params.split)));
292  train_test_split(X,y,X_t,X_v,y_t,y_v,params.shuffle);
293 
294  // define terminals based on size of X
295  params.set_terminals(X.rows());
296 
297  // initial model on raw input
298  params.msg("Fitting initial model", 1);
299  initial_model(X,y);
300  params.msg("Initial score: " + std::to_string(best_score), 1);
301 
302  // initialize population
303  params.msg("Initializing population", 1);
304  p_pop->init(best_ind,params);
305  params.msg("Initial population:\n"+p_pop->print_eqns(),2);
306 
307  // resize F to be twice the pop-size x number of samples
308  F.resize(X_t.cols(),int(2*params.pop_size));
309 
310  // evaluate initial population
311  params.msg("Evaluating initial population",1);
312  p_eval->fitness(*p_pop,X_t,y_t,F,params);
313 
314  vector<size_t> survivors;
315 
316  // main generational loop
317  for (unsigned int g = 0; g<params.gens; ++g)
318  {
319 
320  // select parents
321  params.msg("selection..", 2);
322  vector<size_t> parents = p_sel->select(*p_pop, F, params);
323  params.msg("parents:\n"+p_pop->print_eqns(), 2);
324 
325  // variation to produce offspring
326  params.msg("variation...", 2);
327  p_variation->vary(*p_pop, parents, params);
328  params.msg("offspring:\n" + p_pop->print_eqns(true), 2);
329 
330  // evaluate offspring
331  params.msg("evaluating offspring...", 2);
332  p_eval->fitness(*p_pop, X_t, y_t, F, params, true);
333 
334  // select survivors from combined pool of parents and offspring
335  params.msg("survival...", 2);
336  survivors = p_surv->survive(*p_pop, F, params);
337 
338  // reduce population to survivors
339  params.msg("shrinking pop to survivors...",2);
340  p_pop->update(survivors);
341  params.msg("survivors:\n" + p_pop->print_eqns(), 2);
342 
343  update_best();
344  if (params.verbosity>0) print_stats(g+1);
345  }
346  params.msg("finished",1);
347  params.msg("best training representation: " + best_ind.get_eqn(),1);
348  params.msg("train score: " + std::to_string(best_score), 1);
349  // evaluate population on validation set
350  if (params.split < 1.0)
351  {
352  F_v.resize(X_v.cols(),int(2*params.pop_size));
353  p_eval->fitness(*p_pop, X_v, y_v, F_v, params);
354  initial_model(X_v, y_v); // calculate baseline model validation score
355  update_best(); // get the best validation model
356  }
357  params.msg("best validation representation: " + best_ind.get_eqn(),1);
358  params.msg("validation score: " + std::to_string(best_score), 1);
359  }

361  void Fewtwo::initial_model(MatrixXd& X, VectorXd& y)
362  {
366  bool pass = true;
367  VectorXd yhat = p_ml->out(X,y,params,pass);
368 
369  // set terminal weights based on model
370  params.set_term_weights(p_ml->get_weights());
371 
372  if (params.classification) // assign best score as mean accuracy
373  best_score = (yhat.cast<int>().array() != y.cast<int>().array()).cast<double>().mean();
374  else // assign best score as MSE
375  best_score = (yhat-y).array().pow(2).mean();
376 
377  // initialize best_ind to be all the features
378  best_ind = Individual();
379  for (unsigned i =0; i<X.rows(); ++i)
380  best_ind.program.push_back(params.terminals[i]);
381  best_ind.fitness = best_score;
382  }

421  void Fewtwo::print_stats(unsigned int g)
422  {
423  unsigned num_models = std::min(20,p_pop->size());
424  double med_score = median(F.colwise().mean().array()); // median loss
425  ArrayXd Sizes(p_pop->size()); unsigned i = 0; // collect program sizes
426  for (const auto& p : p_pop->individuals){ Sizes(i) = p.size(); ++i;}
427  unsigned med_size = median(Sizes); // median program size
428  unsigned max_size = Sizes.maxCoeff();
429  string bar, space = ""; // progress bar
430  for (unsigned int i = 0; i<50; ++i){
431  if (i <= 50*g/params.gens) bar += "/";
432  else space += " ";
433  }
434  std::cout.precision(5);
435  std::cout << std::scientific;
436  std::cout << "Generation " << g << "/" << params.gens << " [" + bar + space + "]\n";
437  std::cout << "Min Loss\tMedian Loss\tMedian (Max) Size\tTime (s)\n"
438  << best_score << "\t" << med_score << "\t" ;
439  std::cout << std::fixed << med_size << " (" << max_size << ") \t\t" << timer << "\n";
440  std::cout << "Representation Pareto Front--------------------------------------\n";
441  std::cout << "Rank\tComplexity\tLoss\tRepresentation\n";
442  std::cout << std::scientific;
443  // printing 10 individuals from the pareto front
444  unsigned n = 1;
445  vector<size_t> f = p_pop->sorted_front(n);
446  vector<size_t> fnew(2,0);
447  while (f.size() < num_models && fnew.size()>1)
448  {
449  fnew = p_pop->sorted_front(++n);
450  f.insert(f.end(),fnew.begin(),fnew.end());
451  }
452 
453  for (unsigned j = 0; j < std::min(num_models,unsigned(f.size())); ++j)
454  {
455  std::cout << p_pop->individuals[f[j]].rank << "\t"
456  << p_pop->individuals[f[j]].complexity() << "\t" << (*p_pop)[f[j]].fitness
457  << "\t" << p_pop->individuals[f[j]].get_eqn() << "\n";
458  }
459 
460 
461  // ref counting
462  vector<float> use(params.terminals.size());
463  float use_sum=0;
464  for (unsigned i = 0; i< params.terminals.size(); ++i)
465  {
466  use[i] = float(params.terminals[i].use_count());
467  use_sum += use[i];
468  }
469  vector<size_t> use_idx = argsort(use);
470  std::reverse(use_idx.begin(), use_idx.end());
471 
472  int nf = std::min(5,int(params.terminals.size()));
473  std::cout << "Top " << nf <<" features (\% usage):\n";
474  std::cout.precision(1);
475  for (unsigned i = 0; i<nf; ++i)
476  std::cout << std::fixed << params.terminals[use_idx[i]]->name
477  << " (" << use[use_idx[i]]/use_sum*100 << "\%)\t";
478 
479  std::cout <<"\n\n";
480  }
