160  size_t Individual::subtree(size_t i, char otype='0') const
161  {
162 
178  assert(i>=0 && "attempting to grab subtree with index < 0");
179 
180  if (program[i]->total_arity()==0)
181  {
182  return i;
183  }
184 
185  std::map<char, unsigned int> arity = program[i]->arity;
186 
187  if (otype!='0') // otype=0 means we are at the root. if we are recursing, we need to find
188  // where the nodes to recurse are.
189  {
190  while (program[i-1]->otype != otype && i > 0) --i;
191  assert(program[i]->otype == otype && "invalid subtree arguments");
192  }
193  size_t i2 = i; // index for second recursion
194 
195  for (unsigned int j = 0; j<arity['f']; ++j)
196  i = subtree(--i,'f'); // recurse for floating arguments
197  for (unsigned int j = 0; j<arity['b']; ++j)
198  i2 = subtree(--i2,'b'); // recurse for boolean arguments
199  return std::min(i,i2);
200  }

226  int Individual::check_dominance(const Individual& b) const
227  {
228  /* Check whether this individual dominates b.
229  *
230  * Input:
231  *
232  * b: another individual
233  *
234  * Output:
235  *
236  * 1: this individual dominates b; -1: b dominates this; 0: neither dominates
237  */
238 
239  int flag1 = 0, // to check if this has a smaller objective
240  flag2 = 0; // to check if b has a smaller objective
241 
242  for (int i=0; i<obj.size(); ++i) {
243  if (obj[i] < b.obj[i])
244  flag1 = 1;
245  else if (obj[i] > b.obj[i])
246  flag2 = 1;
247  }
248 
249  if (flag1==1 && flag2==0)
250  // there is at least one smaller objective for this and none for b
251  return 1;
252  else if (flag1==0 && flag2==1)
253  // there is at least one smaller objective for b and none for this
254  return -1;
255  else
256  // no smaller objective or both have one smaller
257  return 0;
258 
259  }
