201  int Individual::check_dominance(const Individual& b) const
202  {
203  /* Check whether this individual dominates b.
204  *
205  * Input:
206  *
207  * b: another individual
208  *
209  * Output:
210  *
211  * 1: this individual domintes b; -1: b dominates this; 0: neither dominates
212  */
213 
214  int flag1 = 0, // to check if this has a smaller objective
215  flag2 = 0; // to check if b has a smaller objective
216 
217  for (int i=0; i<obj.size(); ++i) {
218  if (obj[i] < b.obj[i])
219  flag1 = 1;
220  else if (obj[i] > b.obj[i])
221  flag2 = 1;
222  }
223 
224  if (flag1==1 && flag2==0)
225  // there is at least one smaller objective for this and none for b
226  return 1;
227  else if (flag1==0 && flag2==1)
228  // there is at least one smaller objective for b and none for this
229  return -1;
230  else
231  // no smaller objective or both have one smaller
232  return 0;
