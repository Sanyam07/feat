98  double median(const ArrayXd& v)
99  {
100  // instantiate a vector
101  vector<double> x(v.size());
102  x.assign(v.data(),v.data()+v.size());
103  // middle element
104  size_t n = x.size()/2;
105  // sort nth element of array
106  nth_element(x.begin(),x.begin()+n,x.end());
107  // if evenly sized, return average of middle two elements
108  if (x.size() % 2 == 0) {
109  nth_element(x.begin(),x.begin()+n-1,x.end());
110  return (x[n] + x[n-1]) / 2;
111  }
112  // otherwise return middle element
113  else
114  return x[n];
115  }
