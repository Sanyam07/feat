153  void FeatCV::create_folds(int cols)
154  {
155 
156  cout<<"Creating folds\n";
157 
158  int quantity = cols/foldSize;
159  int remains = cols%foldSize;
160  int i;
161  int startIndex = 0;
162 
163  // determining indexes and umber of datasets in each fold
164  for(i = 0; i < foldSize; i++)
165  {
166  struct DataFolds fold;
167 
168  fold.startIndex = startIndex;
169 
170  if(i+1 <= remains)
171  {
172  fold.quantity = quantity+1;
173  startIndex += quantity+1;
174  }
175  else
176  {
177  fold.quantity = quantity;
178  startIndex += quantity;
179  }
180 
181  dataFolds.push_back(fold);
182 
183  }
184 
185  cout<<"Folds creation complete\n";

188  void FeatCV::fit(MatrixXd x, VectorXd &y)
189  {
190  create_objects();
191  create_folds(x.cols());
192 
193  cout<<"Total number of objects created are "<<featObjs.size()<<"\n";
194 
195  //#pragma omp parallel for
196  for(int i = 0; i < featObjs.size(); i++)
197  {
198  VectorXd objScore(foldSize);
199 
200  int testIndex = foldSize - 1;
201 
202  // loop to fit all data folds in a single object and calculate the mean score
203  for(int j = 0; j < foldSize; j++)
204  {
205  int filled = 0, k, l;
206 
207  MatrixXd trainX(x.rows(), x.cols() - dataFolds[testIndex].quantity);
208  VectorXd trainY(x.cols() - dataFolds[testIndex].quantity);
209 
210  //creating training data from data folds
211  for(k = 0; k < foldSize; k++)
212  {
213  if(k != testIndex)
214  {
215  trainX.block(0, filled, x.rows(), dataFolds[k].quantity) =
216  x.block(0, dataFolds[k].startIndex, x.rows(), dataFolds[k].quantity);
217  filled += dataFolds[k].quantity;
218  }
219  }
220 
221  // creating training data outputs from data folds
222  for(k = 0, l = 0; k < y.size(); k++)
223  if(k < dataFolds[testIndex].startIndex ||
224  k >= dataFolds[testIndex].startIndex+dataFolds[testIndex].quantity)
225  trainY(l++) = y(k);
226 
227  // set dtypes for training data and train the model
228  featObjs[i].obj.set_dtypes(find_dtypes(trainX));
229  featObjs[i].obj.fit(trainX, trainY);
230 
231  // extracting testdata and actual values for test data from data folds
232  MatrixXd testData(x.rows(), dataFolds[testIndex].quantity);
233  VectorXd actualValues(dataFolds[testIndex].quantity);
234 
235  testData <<
236  x.block(0, dataFolds[testIndex].startIndex,x.rows(),dataFolds[testIndex].quantity);
237 
238  for(k = 0; k < dataFolds[testIndex].quantity; k++)
239  actualValues(k) = y(dataFolds[testIndex].startIndex+k);
240 
241  // prediction on test data
242  VectorXd prediction = featObjs[i].obj.predict(testData);
243 
244  // calculating difference between actual and predicted values
245  if (featObjs[i].obj.get_classification())
246  // use classification accuracy
247  objScore[j] = ((prediction.cast<int>().array() !=
248  actualValues.cast<int>().array()).cast<double>()).mean();
249  else
250  // use mean squared error
251  objScore[j] = ((prediction - actualValues).array().pow(2)).mean();
252 
253  testIndex = (testIndex+1)%foldSize;
254 
255  }
256 
257  // setting mean score for the model based on the tuning parameters
258  featObjs[i].score = objScore.mean();
259  cout<<"****Finished object "<<i<<"\n";
260 
261  }
262 
263  bestScoreIndex = 0;
264 
265  // finding the best model from validation and printing its parameters
266  for(int i = 1; i < featObjs.size(); i++)
267  if(featObjs[i].score < featObjs[bestScoreIndex].score)
268  bestScoreIndex = i;
269 
270  cout << "\n********************\nBest tuning parameters for this data is \n";
271  cout << "\nML = " << featObjs[bestScoreIndex].obj.get_ml();
272  cout << "\nPopulation size = " << featObjs[bestScoreIndex].obj.get_pop_size();
273  cout << "\nGenerations = " << featObjs[bestScoreIndex].obj.get_generations();
274  cout << "\nDimensions = " << featObjs[bestScoreIndex].obj.get_max_dim();
275  cout << "\nDepth = " << featObjs[bestScoreIndex].obj.get_max_depth();
276  cout << "\nCross Rate = " <<featObjs[bestScoreIndex].obj.get_cross_rate();
277  cout << "\nScore = " << featObjs[bestScoreIndex].score<<"\n";
278  }
