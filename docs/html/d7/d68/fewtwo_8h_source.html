127  void Fewtwo::fit(MatrixXd& X, VectorXd& y)
128  {
149  // define terminals based on size of X
150  params.set_terminals(X.rows());
151 
152  // initial model on raw input
153  params.msg("Fitting initial model", 1);
154  initial_model(X,y);
155 
156 
157  // initialize population
158  params.msg("Initializing population", 1);
159  p_pop->init(params);
160 
161  // resize F to be twice the pop-size x number of samples
162  F.resize(int(2*params.pop_size),X.cols());
163 
164  // evaluate initial population
165  params.msg("Evaluating initial population",1);
166  p_eval->fitness(*p_pop,X,y,F,params);
167 
168  vector<size_t> survivors;
169 
170  // main generational loop
171  for (size_t g = 0; g<params.gens; ++g)
172  {
173  params.msg("g " + std::to_string(g),1);
174 
175  // select parents
176  params.msg("selection", 1);
177  vector<size_t> parents = p_sel->select(F, params);
178 
179  // variation to produce offspring
180  params.msg("variation", 1);
181  p_variation->vary(*p_pop, parents, params);
182 
183  // evaluate offspring
184  params.msg("evaluation", 1);
185  p_eval->fitness(*p_pop, X, y, F, params);
186 
187  // select survivors from combined pool of parents and offspring
188  params.msg("survival", 1);
189  survivors = p_surv->select(F, params);
190 
191  // reduce population to survivors
192  p_pop->update(survivors);
193  }
194  }
