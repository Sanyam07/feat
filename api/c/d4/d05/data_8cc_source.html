155  void DataRef::train_test_split(bool shuffle, float split)
156  {
157  /* @param X: n_features x n_samples matrix of training data
158  * @param y: n_samples vector of training labels
159  * @param shuffle: whether or not to shuffle X and y
160  * @param[out] X_t, X_v, y_t, y_v: training and validation matrices
161  */
162 
163  // resize training and test sets
164  X_t.resize(o->X.rows(),int(o->X.cols()*split));
165  X_v.resize(o->X.rows(),int(o->X.cols()*(1-split)));
166  y_t.resize(int(o->y.size()*split));
167  y_v.resize(int(o->y.size()*(1-split)));
168 
169  if (shuffle) // generate shuffle index for the split
170  {
171  Eigen::PermutationMatrix<Dynamic,Dynamic> perm(o->X.cols());
172  perm.setIdentity();
173  r.shuffle(perm.indices().data(), perm.indices().data()+perm.indices().size());
174  o->X = o->X * perm; // shuffles columns of X
175  o->y = (o->y.transpose() * perm).transpose() ; // shuffle y too
176 
177  if(o->Z.size() > 0)
178  {
179  std::vector<int> zidx(o->y.size());
180  std::iota(zidx.begin(), zidx.end(), 0);
181  VectorXi zw = Map<VectorXi>(zidx.data(), zidx.size());
182  // shuffle z indices
183  zw = (zw.transpose()*perm).transpose();
184  // assign shuffled zw to zidx
185  zidx.assign(zw.data(), zw.data() + zw.size());
186  for(auto &val : o->Z)
187  {
188  reorder_longitudinal(val.second.first, zidx);
189  reorder_longitudinal(val.second.second, zidx);
190  }
191  }
192 
193  }
194 
195  // map training and test sets
196  t->X = MatrixXf::Map(o->X.data(),t->X.rows(),t->X.cols());
197  v->X = MatrixXf::Map(o->X.data()+t->X.rows()*t->X.cols(),
198  v->X.rows(),v->X.cols());
199 
200  t->y = VectorXf::Map(o->y.data(),t->y.size());
201  v->y = VectorXf::Map(o->y.data()+t->y.size(),v->y.size());
202  if(o->Z.size() > 0)
203  split_longitudinal(o->Z, t->Z, v->Z, split);
204 
205  }
