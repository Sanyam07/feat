380  for (const auto& i : uc)
381  c.push_back(int(i));

383  // sensitivity (TP) and specificity (TN)
384  vector<double> TP(c.size(),0.0), TN(c.size(), 0.0), P(c.size(),0.0), N(c.size(),0.0);
385  ArrayXd class_accuracies(c.size());
386 
387  // get class counts
388 
389  for (unsigned i=0; i< c.size(); ++i)
390  {
391  P.at(i) = (y.array().cast<int>() == c[i]).count(); // total positives for this class
392  N.at(i) = (y.array().cast<int>() != c[i]).count(); // total negatives for this class
393  }
394 
395 
396  for (unsigned i = 0; i < y.rows(); ++i)
397  {
398  if (yhat(i) == y(i)) // true positive
399  ++TP.at(y(i) == -1 ? 0 : y(i)); // if-then ? accounts for -1 class encoding
400 
401  for (unsigned j = 0; j < c.size(); ++j)
402  if ( y(i) !=c.at(j) && yhat(i) != c.at(j) ) // true negative
403  ++TN.at(j);
404 
405  }
406 
407  // class-wise accuracy = 1/2 ( true positive rate + true negative rate)
408  for (unsigned i=0; i< c.size(); ++i){
409  class_accuracies(i) = (TP[i]/P[i] + TN[i]/N[i])/2;

413  }
