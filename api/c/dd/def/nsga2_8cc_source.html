20  size_t NSGA2::tournament(vector<Individual>& pop, size_t i, size_t j) const
21  {
22  Individual& ind1 = pop.at(i);
23  Individual& ind2 = pop.at(j);
24 
25  int flag = ind1.check_dominance(ind2);
26 
27  if (flag == 1) // ind1 dominates ind2
28  return i;
29  else if (flag == -1) // ind2 dominates ind1
30  return j;
31  else if (ind1.crowd_dist > ind2.crowd_dist)
32  return i;
33  else if (ind2.crowd_dist > ind1.crowd_dist)
34  return j;

71  vector<size_t> NSGA2::survive(Population& pop, const MatrixXd& F, const Parameters& params)
72  {
73  /* Selection using the survival scheme of NSGA-II.
74  *
75  * Input:
76  *
77  * pop: population of programs.
78  * params: parameters.
79  * r: random number generator
80  *
81  * Output:
82  *
83  * selected: vector of indices corresponding to pop that are selected.
84  * modifies individual ranks, objectives and dominations.
85  */
86 
87  // set objectives
88  #pragma omp parallel for
89  for (unsigned int i=0; i<pop.size(); ++i)
90  pop.individuals[i].set_obj(params.objectives);
91 
92  // fast non-dominated sort
93  fast_nds(pop.individuals);
94 
95  // Push back selected individuals until full
96  vector<size_t> selected;
97  int i = 0;
98  while ( selected.size() + front[i].size() < params.pop_size)
99  {
100  std::vector<int>& Fi = front[i]; // indices in front i
101  crowding_distance(pop,i); // calculate crowding in Fi
102 
103  for (int j = 0; j < Fi.size(); ++j) // Pt+1 = Pt+1 U Fi
104  selected.push_back(Fi[j]);
105 
106  ++i;
107  }
108 
109  crowding_distance(pop,i); // calculate crowding in final front to include
110  std::sort(front[i].begin(),front[i].end(),sort_n(pop));
111 
112  const int extra = params.pop_size - selected.size();
113  for (int j = 0; j < extra; ++j) // Pt+1 = Pt+1 U Fi[1:N-|Pt+1|]
114  selected.push_back(front[i][j]);
115 
116  return selected;
117  }

119  void NSGA2::fast_nds(vector<Individual>& individuals)
120  {
121  front.resize(1);
122  front[0].clear();
123  //std::vector< std::vector<int> > F(1);
124  #pragma omp parallel for
125  for (int i = 0; i < individuals.size(); ++i) {
126 
127  std::vector<unsigned int> dom;
128  int dcount = 0;
129 
130  Individual& p = individuals[i];
131  // p.dcounter = 0;
132  // p.dominated.clear();
133 
134  for (int j = 0; j < individuals.size(); ++j) {
135 
136  Individual& q = individuals[j];
137 
138  int compare = p.check_dominance(q);
139  if (compare == 1) { // p dominates q
140  //p.dominated.push_back(j);
141  dom.push_back(j);
142  } else if (compare == -1) { // q dominates p
143  //p.dcounter += 1;
144  dcount += 1;
145  }
146  }
147 
148  #pragma omp critical
149  {
150  p.dcounter = dcount;
151  p.dominated.clear();
152  p.dominated = dom;
153 
154 
155  if (p.dcounter == 0) {
156  p.set_rank(1);
157  front[0].push_back(i);
158  }
159  }
160 
161  }
162 
163  // using OpenMP can have different orders in the front[0]
164  // so let's sort it so that the algorithm is deterministic
165  // given a seed
166  std::sort(front[0].begin(), front[0].end());
167 
168  int fi = 1;
169  while (front[fi-1].size() > 0) {
170 
171  std::vector<int>& fronti = front[fi-1];
172  std::vector<int> Q;
173  for (int i = 0; i < fronti.size(); ++i) {
174 
175  Individual& p = individuals[fronti[i]];
176 
177  for (int j = 0; j < p.dominated.size() ; ++j) {
178 
179  Individual& q = individuals[p.dominated[j]];
180  q.dcounter -= 1;
181 
182  if (q.dcounter == 0) {
183  q.set_rank(fi+1);
184  Q.push_back(p.dominated[j]);
185  }
186  }
187  }
188 
189  fi += 1;
190  front.push_back(Q);
191  }

195  void NSGA2::crowding_distance(Population& pop, int fronti)
196  {
197  std::vector<int> F = front[fronti];
198  if (F.size() == 0 ) return;
199 
200  const int fsize = F.size();
201 
202  for (int i = 0; i < fsize; ++i)
203  pop.individuals[F[i]].crowd_dist = 0;
204 
205 
206  const int limit = pop[0].obj.size();
207  for (int m = 0; m < limit; ++m) {
208 
209  std::sort(F.begin(), F.end(), comparator_obj(pop,m));
210 
211  // in the paper dist=INF for the first and last, in the code
212  // this is only done to the first one or to the two first when size=2
213  pop.individuals[F[0]].crowd_dist = std::numeric_limits<double>::max();
214  if (fsize > 1)
215  pop.individuals[F[fsize-1]].crowd_dist = std::numeric_limits<double>::max();
216 
217  for (int i = 1; i < fsize-1; ++i)
218  {
219  if (pop.individuals[F[i]].crowd_dist != std::numeric_limits<double>::max())
220  { // crowd over obj
221  pop.individuals[F[i]].crowd_dist +=
222  (pop.individuals[F[i+1]].obj[m] - pop.individuals[F[i-1]].obj[m])
223  / (pop.individuals[F[fsize-1]].obj[m] - pop.individuals[F[0]].obj[m]);
224  }
225  }
226  }
227  }
228  }
